creating a simple class:

//creating a class
class student{
//class constructor
constructor(name, year){
this.name=name;
this.year=year;
}
//methods(age)
age(){
let date= new Date();
return date.getFullYear()-this.year;
}
}

class inheritance:

//parent class and child class(class inheritance)
class phone{
constructor(brand){
this.phonename=brand;
}
myphone(){
return 'i have a ' +this.phonename;
}
}

class model extends phone{
constructor(brand, model){
super(brand);
this.model=model;
}
show(){
return this.myphone() + ', it is a ' +this.model;
}
}
